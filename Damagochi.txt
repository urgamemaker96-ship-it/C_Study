int 메인함수(void)
{
	int 체력 = 0;
	int 포만도 = 0;
	int 배변활동 = 50;
	int 입력받는변수;

	반복문(1)
	{
		if(배변활동 >= 100)
		{
			다마고치병걸렸다.
			게임오버됐다.
			종료제어문
		}
		상자모양 테두리
		상태창
		체력출력
		포만도출력
		배변활동출력
		상자모양 테두리
	

		상자모양 테두리
		1.잠자기 2.응가하기 3.밥주기 출력
		상자모양 테두리
		입력: 출력

		입력받는변수 입력


		분기문(입력받을변수)
		1번 : 
			if(체력 >= 100)
			{
				더이상 잘필요가 없다.

			}
			else
			{
				체력 = 체력+10
				다마고치 체력 10 증가 출력
			}
			종료제어문

		2번 :
			배변활동 = 0;
			다마고치 배변활동 0 출력
			종료제어문
		3번:
			if(포만도 >= 200)
			{
				더이상 먹을필요없다 출력
			
			}
			else
			{
				포만도 = 포만도+20;
				배변활동 = 배변활동 + 20;
				다마고치 포만도 20 증가했다 출력
			}
			종료제어문

	}

}

// 1단계
1. 체력 , 포만도, 배변활동 선언 후 값 초기화, 입력 받을 변수를 선언
2. while 반목문을 이용, 해당 블록에서 분기문 작성 및 병걸릴 조건과 게임오버 조건 작성
